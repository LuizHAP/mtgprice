class CircuitBreaker<TI extends unknown[] = unknown[], TR = unknown> extends EventEmitter {
  closed: boolean      // true when circuit allows requests
  opened: boolean      // true when circuit is open (rejecting)
  halfOpen: boolean    // true when testing recovery
  stats: {
    failures: number
    successes: number
    fallbacks: number
    rejects: number
    fires: number
    timeouts: number
    cacheHits: number
    cacheMisses: number
    semaphoreRejections: number
  }
  fire(...args: TI): Promise<TR>
  fallback(fn: (...args: TI) => unknown): this
  shutdown(): void                         // clears timers — call in afterEach to prevent leaks
  on(event: 'open' | 'close' | 'halfOpen' | 'fallback' | 'failure' | 'success', handler: Function): this
}
