```typescript
import type { DetectionConfig } from './config'

export interface EvaluateInput {
  latest: number | null
  baseline: number | null
  sevenDayAgo: number | null
  historyDays: number
}

export interface EvaluateResult {
  fires: boolean
  dropPercent: number | null
  reason?: 'above_baseline' | 'insufficient_drop' | 'cold_start' | 'no_lookback_price' | 'no_current_price' | 'no_baseline'
}

export function evaluateCandidate(input: EvaluateInput, config: DetectionConfig): EvaluateResult {
  if (input.historyDays < config.minHistoryDays) return { fires: false, dropPercent: null, reason: 'cold_start' }
  if (input.latest === null) return { fires: false, dropPercent: null, reason: 'no_current_price' }
  if (input.baseline === null) return { fires: false, dropPercent: null, reason: 'no_baseline' }
  if (input.sevenDayAgo === null || input.sevenDayAgo <= 0) return { fires: false, dropPercent: null, reason: 'no_lookback_price' }

  const dropPercent = Math.round(((input.sevenDayAgo - input.latest) / input.sevenDayAgo) * 10000) / 100
  const meetsBaseline = input.latest <= input.baseline
  const meetsDrop = input.latest <= input.sevenDayAgo * (1 - config.dropThreshold)

  if (!meetsBaseline) return { fires: false, dropPercent, reason: 'above_baseline' }
  if (!meetsDrop) return { fires: false, dropPercent, reason: 'insufficient_drop' }
  return { fires: true, dropPercent }
}
```

    D-07 CANDIDATE STATE MACHINE (implements CONTEXT.md D-07 literally)

    For each `(card_id, source)` in (wishlist × 4 sources), the orchestrator runs this
    4-step flow. It is the ONLY correct reading of D-07: "the criteria must be met on
    two consecutive collection runs for the same (card_id, source) before an alert fires."

    Preconditions for entering the flow (if any fail, SKIP the pair entirely — do not
    touch detection_candidates):
      - `await isInCooldown(cardId, source, config)` must be false. A pair in cooldown
        is silent regardless of candidate state.

    Pair-level evaluation:

      const result = evaluateCandidate(
        { latest, baseline, sevenDayAgo, historyDays },
        config,
      )

    Step 1. `result.fires === false`:
      - This run does NOT satisfy the D-01/D-02/D-03/D-04/D-06 check.
      - `await deleteCandidate(cardId, source)` — the streak (if any) is BROKEN.
      - Do NOT emit an opportunity.
      - Continue to the next (card, source) pair.

    Step 2. `result.fires === true`:
      - This run DOES satisfy the check. Look up the existing candidate:

        const existing = await getCandidate(cardId, source)

      2a. `existing === null`:
        - This is the FIRST consecutive confirming run for the pair.
        - `await insertCandidate(cardId, source)` — record that this pair was confirmed
          in the current run. `first_seen_at` defaults to now().
        - Do NOT emit an opportunity. D-07 requires TWO consecutive runs; this is run 1.
        - Continue to the next pair.

      2b. `existing !== null`:
        - The previous run ALSO confirmed this pair (because the candidate row still
          exists — if the previous run had not confirmed it, Step 1 would have deleted
          the row). Therefore this is the SECOND consecutive confirming run.
        - PROMOTE: push the DetectedOpportunity to the result array:

          results.push({
            cardId,
            cardName: card.name,
            source,
            currentPrice: latest,
            baselinePrice: baseline,
            sevenDayAgoPrice: sevenDayAgo,
            dropPercent: result.dropPercent!,
          })

        - `await deleteCandidate(cardId, source)` — the candidate has been consumed;
          the pair now moves into the cooldown window (when Plan 04 persists the
          opportunities row).
        - The caller (Plan 04) handles the actual INSERT into `opportunities` plus
          the cooldown check via `isInCooldown` on the NEXT run.

    Step 3. Stale housekeeping (once per run, BEFORE iterating over pairs):

      const staleDeleted = await deleteStaleCandidates()
      logger.debug(`D-07 housekeeping: deleted ${staleDeleted} stale detection_candidates rows`)

      This deletes rows where `first_seen_at < now() - interval '2 days'`. Rationale:
      if a card temporarily disappears from the scrape (network failure, provider
      outage), its candidate row would linger forever without this sweep. 2 days >
      (2 cron runs + buffer) is the minimum safe window.

    Step 4. Logging:

      logger.info(
        `Detection complete: wishlist=${wishlist.length} cards, eligible pairs=${eligiblePairs}, promotions=${results.length}`,
      )

      Do NOT log userId, prices, card names, or source-per-pair at info level.
      Use debug for per-pair prices and per-pair decisions.

    RACE CONDITIONS:
    - `detectOpportunitiesForWishlist` is called from `executePriceCollection` which is
      already single-flighted by the `isRunning` guard in `src/scheduler/jobs.ts`.
      Therefore at most one instance of this state machine runs at a time and there is
      no read-modify-write race on `detection_candidates`.
    - The UNIQUE (card_id, source) constraint on `detection_candidates` provides a
      last-line defense: `insertCandidate` uses `ON CONFLICT DO NOTHING` semantics so
      a retry within the same run is idempotent.
  </state_machine>

    userId: number,
    config: DetectionConfig,
  ): Promise<DetectedOpportunity[]> {
    // Step 3 (run once, before the loop): stale housekeeping
    const staleDeleted = await deleteStaleCandidates()
    logger.debug(`D-07 housekeeping: deleted ${staleDeleted} stale detection_candidates rows`)

    const wishlist = await getUserWishlist(userId)
    const results: DetectedOpportunity[] = []
    let eligiblePairs = 0

    const SOURCES = ['ligamagic', 'tcgplayer', 'cardmarket', 'cardkingdom'] as const

    for (const card of wishlist) {
      for (const source of SOURCES) {
        // Preconditions: cooldown check
        if (await isInCooldown(card.oracleId, source, config)) continue
        eligiblePairs += 1

        const [latest, baseline, sevenDayAgo, historyDays] = await Promise.all([
          getLatestPrice(card.oracleId, source),
          getBaselineMean(card.oracleId, source, config),
          getPriceSevenDaysAgo(card.oracleId, source, config),
          getHistoryDaysForPair(card.oracleId, source),
        ])

        const result = evaluateCandidate({ latest, baseline, sevenDayAgo, historyDays }, config)

        if (!result.fires) {
          // Step 1: streak broken (or never started) — delete any existing candidate
          await deleteCandidate(card.oracleId, source)
          continue
        }

        // Step 2: candidate confirmed this run
        const existing = await getCandidate(card.oracleId, source)
        if (existing === null) {
          // 2a: first confirming run — record candidate, do NOT alert
          await insertCandidate(card.oracleId, source)
          logger.debug(`D-07 candidate created for ${card.oracleId}/${source}`)
          continue
        }

        // 2b: second consecutive confirming run — PROMOTE
        results.push({
          cardId: card.oracleId,
          cardName: card.name,
          source,
          currentPrice: latest!,
          baselinePrice: baseline!,
          sevenDayAgoPrice: sevenDayAgo!,
          dropPercent: result.dropPercent!,
        })
        await deleteCandidate(card.oracleId, source)
        logger.debug(`D-07 promotion for ${card.oracleId}/${source}`)
      }
    }

    logger.info(
      `Detection complete: wishlist=${wishlist.length} cards, eligible pairs=${eligiblePairs}, promotions=${results.length}`,
    )
    return results
  }
